class Solution {
public:
    int totalMoney(int n) {

        int amount=0;
        int start=1;
        int count=0;

        for(int i=start;count<n;i++){
            amount+=(i);
            // cout<<i<<" "; 
            count++;
            if(i==start+6){
                start++;
                i=start-1;
            }
        }
        return amount;
        
    }
};
